    constructor(cellW, cellH, ncol) {
        this.pos = createVector(random(width), random(height));
        this.vel = createVector(0, 0);
        this.acc = createVector(0, 0);
        this.maxSpeed = 4;
        this.prevPos = this.pos.copy();
        this.cellW = cellW;
        this.cellH = cellH;
        this.ncol = ncol;
    }

    update() {
        this.vel.add(this.acc);
        this.vel.limit(this.maxSpeed);
        this.pos.add(this.vel);
        this.acc.mult(0);
    }

    follow(vec) {
        var x = floor(this.pos.x / this.cellW);
        var y = floor(this.pos.y / this.cellH);
        var index = x + y * this.ncol;
        var force = vec[index];
        this.applyforce(force);
    }

    applyforce(force) {
        this.acc.add(force);
    }

    getcolor() {
        let color = strokeCP
            .color()
            ._array.slice(0, 3)
            .concat(opacitySlide.value())
            .map((d) => d * 255);
        return color;
    }

    show() {
        stroke(this.getcolor());
        strokeWeight(0.6);
        strokeJoin(ROUND);
        line(this.pos.x, this.pos.y, this.prevPos.x, this.prevPos.y);
        this.updateprev();
    }

    updateprev() {
        this.prevPos = this.pos.copy();
    }

    edges() {
        if (this.pos.x > width || this.pos.x < 0 || this.pos.y > height || this.pos.y < 0) {
            this.pos = createVector(random(width), random(height));
            this.updateprev();
        }
    }
